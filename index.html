-- OOP Combat Module
local CombatSystem = {}
CombatSystem.__index = CombatSystem

function CombatSystem.new(player)
  local self = setmetatable({},
    CombatSystem)
  self.player  = player
  self.damage  = 25
  self.cooldown = 0.6
  self._active = false
  return self
end

function CombatSystem:attack()
  if self._active then return end
  self._active = true
  -- fire hitbox, apply damage
  task.delay(self.cooldown, function()
    self._active = false
  end)
end
